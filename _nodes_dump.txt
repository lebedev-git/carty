// Вспомогательная функция для очистки промпта от оберток ```text или ```markdown
function cleanPrompt(text) {
  if (!text) return "";
  return text
    .replace(/^```(markdown|text|json)?\s*/i, '') // Убираем начало ```text
    .replace(/^```\s*/, '')                        // Убираем просто ```
    .replace(/\s*```$/, '')                        // Убираем конец ```
    .trim();
}

const results = inputData.map(item => {
  let title = "";
  let number = "";
  let prompt = "";

  // === ВАРИАНТ 1: Элемент массива — СТРОКА (Markdown) ===
  if (typeof item === 'string') {
    // Сценарий А: Заголовок СНАРУЖИ (Ваш последний пример: "Текст №1...\n\n```text...")
    // Ищем текст до ```, затем содержимое внутри ```
    const matchOutside = item.match(/^(.*?)\s*```(?:text|markdown)?\s*([\s\S]*?)\s*```/);
    
    // Сценарий Б: Заголовок ВНУТРИ (Ваш предыдущий пример: "```markdown\nТекст №1...\n\n...")
    // Ищем начало блока, потом заголовок, два переноса, потом промпт
    const matchInside = item.match(/^```(?:markdown|text)?\s*(.*?)\n\n([\s\S]*?)\s*```/);

    if (matchOutside && !item.trim().startsWith('```')) {
      // Сработал Сценарий А (заголовок перед кодом)
      title = matchOutside[1].trim();
      prompt = matchOutside[2].trim();
    } else if (matchInside) {
      // Сработал Сценарий Б (заголовок внутри кода)
      title = matchInside[1].trim();
      prompt = matchInside[2].trim();
    } else {
      // Если регулярки не сработали, просто чистим текст
      prompt = cleanPrompt(item);
    }
  } 
  
  // === ВАРИАНТ 2: Элемент массива — ОБЪЕКТ (JSON) ===
  else if (typeof item === 'object' && item !== null) {
    if ('card_id' in item || 'header' in item) {
      // Формат с card_id
      title = item.header || "";
      number = item.card_id || "";
      prompt = cleanPrompt(item.prompt_code);
    } else {
      // Формат с "Номер" и "Промт"
      title = ""; 
      number = item["Номер"] || "";
      prompt = cleanPrompt(item["Промт"]);
    }
  }

  // === ФИНАЛЬНАЯ ОБРАБОТКА ===
  // Если номер не найден в явном поле (card_id/Номер), вытаскиваем его из начала текста промпта
  if (!number && prompt) {
    const numMatch = prompt.match(/^(\d+)/);
    number = numMatch ? numMatch[1] : "";
  }

  return {
    json: {
      title: title,
      number: number,
      prompt: prompt
    }
  };
});

return results;

Правила форматирования:

ЗАГОЛОВКИ:
Основной заголовок документа: ## НАЗВАНИЕ ДОКУМЕНТА

Подзаголовки разделов: ### Название раздела

Обязательно оставляй пустую строку после заголовков

ВЫДЕЛЕНИЕ ТЕКСТА:
Жирный текст для важных данных: **текст**

Курсив для примечаний: *текст*

Не злоупотребляй выделением - только ключевые цифры и термины

СПИСКИ:
Маркированные списки: используй дефис с пробелом (- пункт)

Нумерованные списки: используй цифры с точкой (1. пункт)

Каждый пункт с новой строки

Один уровень вложенности (без подпунктов)

ТАБЛИЦЫ:
Используй Markdown формат таблиц:

| Заголовок 1 | Заголовок 2 | Заголовок 3 |
|-------------|-------------|-------------|
| Данные 1    | Данные 2    | Данные 3    |

Первая строка - заголовки

Вторая строка - разделитель (дефисы)

Последующие строки - данные
